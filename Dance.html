from PIL import Image, ImageDraw, ImageFont import textwrap # Create image width, height = 1080, 6000 image = Image.new('RGB', (width, height), color='white') draw = ImageDraw.Draw(image) # Add background gradient for i in range(height): r = int(255 - (i/height)*100) g = int(245 - (i/height)*80) b = int(235 - (i/height)*60) draw.line([(0, i), (width, i)], fill=(r, g, b)) # Add header header_height = 200 draw.rectangle([(0, 0), (width, header_height)], fill='#FF9933') draw.rectangle([(0, header_height-50), (width, header_height)], fill='#138808') draw.rectangle([(0, header_height), (width, header_height+10)], fill='#FFFFFF') # Try to add Hindi font (you may need to download one) try: font_large = ImageFont.truetype("NotoSansDevanagari-Regular.ttf", 50) font_medium = ImageFont.truetype("NotoSansDevanagari-Regular.ttf", 35) font_small = ImageFont.truetype("NotoSansDevanagari-Regular.ttf", 28) except: font_large = ImageFont.load_default() font_medium = ImageFont.load_default() font_small = ImageFont.load_default() # Title title = "भारत के प्रसिद्ध लोक नृत्य" draw.text((width//2, 80), title, fill='white', font=font_large, anchor='mm') # Dance data dances = [ ("1. बसंत", "उत्तराखंड / हिमाचल प्रदेश"), ("2. डंडानटा", "आंध्र प्रदेश"), # ... add all dances here ] # Add dances y_position = 250 for i in range(min(20, len(dances))): # First 20 for example # Dance name draw.text((100, y_position), dances[i][0], fill='#2c3e50', font=font_medium) # State draw.text((400, y_position), dances[i][1], fill='#138808', font=font_small) y_position += 70 # Save image image.save("indian_folk_dances.png") print("Image saved as 'indian_folk_dances.png'")
